def sum_rec(x, n):
    if n == 1:
        return x / 1
    def pow_rec(a, b):
        if b == 1:
            return a
        return a * pow_rec(a, b - 1)
    def fact_rec(k):
        if k == 1:
            return 1
        return k * fact_rec(k - 1)

    return pow_rec(x, n) / fact_rec(n) + sum_rec(x, n - 1)
